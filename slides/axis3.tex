      \draw[ultra thick,<->] ($(problem)+(0,-0.7)$) -- ($(dual)+(0,0.55)$);
      %
      \node[align=center,text width=0.3\textwidth] (optcond) at (3.5,2.4) {
        \begin{blockcolor}{mLightBrown}{Opt. conditions}
            \centering
            $\transp{\dici{\idxentry}}\opt{\dv} \in \subdiff\separable{\relaxrfunc}{\idxentry}(\opt{\pvi{\idxentry}})$
        \end{blockcolor}
      };
      %
      \node (dual-tests) at (0,0.25) {\textbf{Intermediate result}};
      \node[text width=0.9\linewidth,align=center,inner sep=2pt] at ($(dual-tests.south)+(0,-0.4)$) (result-text) {Zeros and non-zeros in any solution $\opt{\pv}$ can be identified from $\opt{\dv}$.};
      \draw[ultra thick] ($(result-text.center)+(5,0.4)$) rectangle ($(result-text.center)+(-5,-0.4)$);
      %
      \node[text width=0.93\linewidth,align=left] (theorem) at ($(result-text)+(0,-2)$) {
        \textbf{Theorem --} One has
        \begin{alignat*}{4}
          \transp{\dici{\idxentry}}\opt{\dv} &\in \interior(\subdiff\separable{\relaxrfunc}{\idxentry}(0)) &&\implies \opt{\pvi{\idxentry}} = 0 \\
          \transp{\dici{\idxentry}}\opt{\dv} &\notin \subdiff\separable{\relaxrfunc}{\idxentry}(0) &&\implies \opt{\pvi{\idxentry}} \neq 0
        \end{alignat*}
        for any solution $\opt{\pv}$.
      };
    \end{scope}
  \end{tikzpicture}
